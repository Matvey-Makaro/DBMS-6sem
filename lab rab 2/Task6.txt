CREATE OR REPLACE TRIGGER correct_c_val_insert
AFTER INSERT
ON STUDENTS
FOR EACH ROW
BEGIN
UPDATE GROUPS SET C_VAL = C_VAL + 1 WHERE ID = :NEW.GROUP_ID;
END correct_c_val_insert;

CREATE OR REPLACE TRIGGER correct_c_val_update
AFTER UPDATE
ON STUDENTS
FOR EACH ROW
BEGIN
UPDATE GROUPS SET C_VAL = C_VAL - 1 WHERE ID = :OLD.GROUP_ID;
UPDATE GROUPS SET C_VAL = C_VAL + 1 WHERE ID = :NEW.GROUP_ID;
END correct_c_val_update;

CREATE OR REPLACE TRIGGER correct_c_val_delete
AFTER DELETE
ON STUDENTS
FOR EACH ROW
BEGIN
UPDATE GROUPS SET C_VAL = C_VAL - 1 WHERE ID = :OLD.GROUP_ID;
END correct_c_val_delete;
